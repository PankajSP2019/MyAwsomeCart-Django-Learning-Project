<script>
    console.log("Working Check")
    // Create cart if not exist, if exists then fetch it
    if(localStorage.getItem('cart') == null){
        var cart = {};
    }else{
        cart = JSON.parse(localStorage.getItem('cart'));
        //! Here 'cart' is navbar cart button ID, cart is variable we created in localStorage
        // It will show how many differnt product in the cart
        document.getElementById('cart').innerHTML = Object.keys(cart).length;
    }

    //!JQuery
    $('.cart').click(function(){
        // After click this funtion will the increse the number of same product, if the product is exists in the cart
        // simply in count the product quantity 
        console.log('Clicked - Checked')
        var idstr = this.id.toString();
        if(cart[idstr] != undefined){
            cart[idstr] = cart[idstr] + 1;
        }else{
            cart[idstr] = 1;
        }

        // Update cart
        console.log(cart);
        localStorage.setItem('cart', JSON.stringify(cart));
        //! Here 'cart' is navbar cart button ID, cart is variable we created in localStorage
        // It will show how many differnt product in the cart
        document.getElementById('cart').innerHTML = Object.keys(cart).length;
    });

</script>
